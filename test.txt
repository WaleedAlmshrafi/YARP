app.MapGet("/Lookup/{**path}", () =>

        List<Lookup> lookups = new List<Lookup>();
        for (var i = 0; i < 15; i++)
        {
            var lookup = new Lookup()
            {
                Id = i,
                NameEn = faker.Company.CompanyName(),
                NameAr = faker.Name.FirstName()
            };
            lookups.Add(lookup);
        }

        var response = new LookupResponse()
        {
            IsSuccess = true,
            Data = lookups,
            Errors = null
        };
        return Results.Ok(response);
    })
    .WithName("Lookups");

app.MapGet("/Parcel/GetByDate", (DateOnly start,DateOnly end,int pageNumber,int pageSize) =>
    {
        Faker faker = new Faker();
        var totalIds = new List<int>();
        for (int i = 0; i < 100; i++) // Let's assume 100 total items
        {
            totalIds.Add(faker.Random.Int(1, 10000));
        }

        // Calculate paging
        int totalItems = totalIds.Count;
        int totalPages = (int)Math.Ceiling(totalItems / (double)pageSize);

        // Apply paging
        var pagedIds = totalIds
            .Skip((pageNumber - 1) * pageSize)
            .Take(pageSize)
            .ToList();

        // Create and return your ParcelIds object
        var result = new ParcelIds
        {
            Ids = pagedIds,
            TotalPages = totalPages
        };

        return Results.Ok(result);
        
    })
    .WithName("ParcelGetByDate");
